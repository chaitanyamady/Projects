SELECT p.product_id, p.product_name
FROM product p
LEFT JOIN (
 SELECT product_id, MAX(sale_date) AS last_sale_date
 FROM sales
 GROUP BY product_id
) s ON p.product_id = s.product_id
WHERE s.last_sale_date IS NULL OR s.last_sale_date < DATE_SUB(NOW(),
INTERVAL 14 DAY);
